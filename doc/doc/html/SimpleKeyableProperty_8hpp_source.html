230 template<typename T>
231 T SimpleKeyableProperty<T>::interpolate(long double pos) const{
232  T best = mValues.begin()->second;
233  for(auto iter = mValues.begin(); iter != mValues.end(); iter++){
234  if(iter->first <= pos){
235  best = iter->second;
236  }
237  else{
238  iter--;
239  long double pos1 = fabs(iter->first - pos);
240  iter++;
241  long double pos2 = fabs(iter->first - pos);
242  long double div = pos1 + pos2;
243  pos1 = 1 - (pos1 / div);
244  pos2 = 1 - (pos2 / div);
245  T next = iter->second;
246  best = best * pos1 + next * pos2;
247  return best;
248  }
249  }
250  return best;
251 }
